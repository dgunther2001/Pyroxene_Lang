152 class binary_expr : public top_level_expr {
153 private:
154 lexer::Token_Type op;
155 std::unique_ptr<top_level_expr> left, right;
156 types type;
157
158 public:
159 binary_expr(lexer::Token_Type op, std::unique_ptr<top_level_expr> left, std::unique_ptr<top_level_expr> right, types type) :
160 op(op),
161 left(std::move(left)),
162 right(std::move(right)),
163 type(type)
164 {}
165
166 types get_expr_type() const override {return type;}
167 void debug_output();
168 char get_op() {
169 switch (op) {
170 case lexer::tok_plus:
171 return '+';
172 break;
173 case lexer::tok_minus:
174 return '-';
175 break;
176 case lexer::tok_mult:
177 return '*';
178 break;
179 case lexer::tok_div:
180 return '/';
181 break;
182 default:
183 std::abort();
184 }
185 }
186 llvm::Value* codegen() override;
187 };
