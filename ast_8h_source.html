166 class binary_expr : public top_level_expr {
167 private:
168 lexer::Token_Type op;
169 std::unique_ptr<top_level_expr> left, right;
170 type_enum::types type;
171
172 public:
173 binary_expr(lexer::Token_Type op, std::unique_ptr<top_level_expr> left, std::unique_ptr<top_level_expr> right) :
174 op(op),
175 left(std::move(left)),
176 right(std::move(right))
177 {}
178
179 void semantic_analysis() override;
180 std::string get_ast_class() const override { return "binary"; }
181 type_enum::types get_expr_type() const override {return type;}
182 void set_expr_type(type_enum::types new_type) { type = new_type; }
183 void debug_output();
184 char get_op() {
185 switch (op) {
186 case lexer::tok_plus:
187 return '+';
188 break;
189 case lexer::tok_minus:
190 return '-';
191 break;
192 case lexer::tok_mult:
193 return '*';
194 break;
195 case lexer::tok_div:
196 return '/';
197 break;
198 default:
199 std::abort();
200 }
201 }
202 llvm::Value* codegen() override;
203 };
