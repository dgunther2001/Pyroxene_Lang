185 private:
186 lexer::Token_Type op;
187 std::unique_ptr<top_level_expr> left, right;
188 type_enum::types type;
189
190 public:
191 binary_expr(lexer::Token_Type op, std::unique_ptr<top_level_expr> left, std::unique_ptr<top_level_expr> right) :
192 op(op),
193 left(std::move(left)),
194 right(std::move(right))
195 {}
196
197 void semantic_analysis() override;
198 std::string get_ast_class() const override { return "binary"; }
199 type_enum::types get_expr_type() const override {return type;}
200 void set_expr_type(type_enum::types new_type) override { type = new_type; }
201 void debug_output();
202 char get_op() {
203 switch (op) {
204 case lexer::tok_plus:
205 return '+';
206 break;
207 case lexer::tok_minus:
208 return '-';
209 break;
210 case lexer::tok_mult:
211 return '*';
212 break;
213 case lexer::tok_div:
214 return '/';
215 break;
216 default:
217 std::abort();
218 }
219 }
220 llvm::Value* codegen() override;
221 };
