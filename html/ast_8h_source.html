190
191 public:
192 binary_expr(lexer::Token_Type op, std::unique_ptr<top_level_expr> left, std::unique_ptr<top_level_expr> right) :
193 op(op),
194 left(std::move(left)),
195 right(std::move(right))
196 {}
197
198 void semantic_analysis() override;
199 std::string get_ast_class() const override { return "binary"; }
200 type_enum::types get_expr_type() const override {return type;}
201 void set_expr_type(type_enum::types new_type) override { type = new_type; }
202 void debug_output();
203 char get_op() {
204 switch (op) {
205 case lexer::tok_plus:
206 return '+';
207 break;
208 case lexer::tok_minus:
209 return '-';
210 break;
211 case lexer::tok_mult:
212 return '*';
213 break;
214 case lexer::tok_div:
215 return '/';
216 break;
217 default:
218 std::abort();
219 }
220 }
