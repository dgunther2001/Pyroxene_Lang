168 class binary_expr : public top_level_expr {
169 private:
170 lexer::Token_Type op;
171 std::unique_ptr<top_level_expr> left, right;
172 type_enum::types type;
173
174 public:
175 binary_expr(lexer::Token_Type op, std::unique_ptr<top_level_expr> left, std::unique_ptr<top_level_expr> right) :
176 op(op),
177 left(std::move(left)),
178 right(std::move(right))
179 {}
180
181 void semantic_analysis() override;
182 std::string get_ast_class() const override { return "binary"; }
183 type_enum::types get_expr_type() const override {return type;}
184 void set_expr_type(type_enum::types new_type) override { type = new_type; }
185 void debug_output();
186 char get_op() {
187 switch (op) {
188 case lexer::tok_plus:
189 return '+';
190 break;
191 case lexer::tok_minus:
192 return '-';
193 break;
194 case lexer::tok_mult:
195 return '*';
196 break;
197 case lexer::tok_div:
198 return '/';
199 break;
200 default:
201 std::abort();
202 }
203 }
204 llvm::Value* codegen() override;
205 };
