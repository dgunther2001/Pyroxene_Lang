169 class binary_expr : public top_level_expr {
170 private:
171 lexer::Token_Type op;
172 std::unique_ptr<top_level_expr> left, right;
173 types type;
174
175 public:
176 binary_expr(lexer::Token_Type op, std::unique_ptr<top_level_expr> left, std::unique_ptr<top_level_expr> right, types type) :
177 op(op),
178 left(std::move(left)),
179 right(std::move(right)),
180 type(type)
181 {}
182
183 void semantic_analysis() override;
184 std::string get_ast_class() const override { return "binary"; }
185 types get_expr_type() const override {return type;}
186 void debug_output();
187 char get_op() {
188 switch (op) {
189 case lexer::tok_plus:
190 return '+';
191 break;
192 case lexer::tok_minus:
193 return '-';
194 break;
195 case lexer::tok_mult:
196 return '*';
197 break;
198 case lexer::tok_div:
199 return '/';
200 break;
201 default:
202 std::abort();
203 }
204 }
205 llvm::Value* codegen() override;
206 };
