183 type_enum::types type;
184
185 public:
186 binary_expr(lexer::Token_Type op, std::unique_ptr<top_level_expr> left, std::unique_ptr<top_level_expr> right) :
187 op(op),
188 left(std::move(left)),
189 right(std::move(right))
190 {}
191
192 void semantic_analysis() override;
193 std::string get_ast_class() const override { return "binary"; }
194 type_enum::types get_expr_type() const override {return type;}
195 void set_expr_type(type_enum::types new_type) override { type = new_type; }
196 void debug_output();
197 char get_op() {
198 switch (op) {
199 case lexer::tok_plus:
200 return '+';
201 break;
202 case lexer::tok_minus:
203 return '-';
204 break;
205 case lexer::tok_mult:
206 return '*';
207 break;
208 case lexer::tok_div:
209 return '/';
210 break;
211 default:
212 std::abort();
213 }
214 }
215 llvm::Value* codegen() override;
