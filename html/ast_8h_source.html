171 class binary_expr : public top_level_expr {
172 private:
173 lexer::Token_Type op;
174 std::unique_ptr<top_level_expr> left, right;
175 types type;
176
177 public:
178 binary_expr(lexer::Token_Type op, std::unique_ptr<top_level_expr> left, std::unique_ptr<top_level_expr> right, types type) :
179 op(op),
180 left(std::move(left)),
181 right(std::move(right)),
182 type(type)
183 {
184 semantic_analysis();
185 }
186
187 void semantic_analysis() override;
188 std::string get_ast_class() const override { return "binary"; }
189 types get_expr_type() const override {return type;}
190 void debug_output();
191 char get_op() {
192 switch (op) {
193 case lexer::tok_plus:
194 return '+';
195 break;
196 case lexer::tok_minus:
197 return '-';
198 break;
199 case lexer::tok_mult:
200 return '*';
201 break;
202 case lexer::tok_div:
203 return '/';
204 break;
205 default:
206 std::abort();
207 }
208 }
209 llvm::Value* codegen() override;
210 };
